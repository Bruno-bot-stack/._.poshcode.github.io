## Sort-Random.ps1
## Shuffle pipeline input into a random order...
## Note that, as with any Sort- function, this is a natural holdup in your pipeline
begin {
  $list = @()
}
process {
  $list += $_
}
end {
  $r = new-object Random
  1..$list.Count | % { $list[$r.Next(0,$list.Count-1)] }
}
